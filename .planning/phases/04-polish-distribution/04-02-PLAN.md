    ```rust
    #[cfg(target_os = "windows")]
    pub fn is_enabled() -> anyhow::Result<bool> {
        use winreg::enums::{HKEY_CURRENT_USER, KEY_READ};
        use winreg::RegKey;

        let hkcu = RegKey::predef(HKEY_CURRENT_USER);
        match hkcu.open_subkey_with_flags(RUN_KEY, KEY_READ) {
            Ok(key) => match key.get_value::<String, _>(VALUE_NAME) {
                Ok(_) => Ok(true),
                Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(false),
                Err(e) => Err(e.into()),
            },
            Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(false),
            Err(e) => Err(e.into()),
        }
    }

    #[cfg(target_os = "windows")]
    pub fn enable() -> anyhow::Result<()> {
        use winreg::enums::HKEY_CURRENT_USER;
        use winreg::RegKey;

        let exe = std::env::current_exe()?;
        let value = format_run_value(&exe);
        let hkcu = RegKey::predef(HKEY_CURRENT_USER);
        let (key, _disp) = hkcu.create_subkey(RUN_KEY)?;
        key.set_value(VALUE_NAME, &value)?;
        tracing::info!(value = %value, "autostart enabled (HKCU\\...\\Run\\Hallmark)");
        Ok(())
    }

    #[cfg(target_os = "windows")]
    pub fn disable() -> anyhow::Result<()> {
        use winreg::enums::{HKEY_CURRENT_USER, KEY_SET_VALUE};
        use winreg::RegKey;

        let hkcu = RegKey::predef(HKEY_CURRENT_USER);
        match hkcu.open_subkey_with_flags(RUN_KEY, KEY_SET_VALUE) {
            Ok(key) => match key.delete_value(VALUE_NAME) {
                Ok(()) => {
                    tracing::info!("autostart disabled (HKCU\\...\\Run\\Hallmark removed)");
                    Ok(())
                }
                Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
                    tracing::debug!("autostart::disable: value already absent");
                    Ok(())
                }
                Err(e) => Err(e.into()),
            },
            Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
                tracing::debug!("autostart::disable: Run key absent");
                Ok(())
            }
            Err(e) => Err(e.into()),
        }
    }
    ```

    ```rust
        fn enable_then_is_enabled_round_trip() {
    ```

    Why double-quote the exe path: Windows installations may live at paths with spaces (`C:\Users\First Last\...`). Quoting prevents the shell argv splitter from interpreting `First Last` as two tokens. RESEARCH Pattern 2 lines 482-486 documents this.

    Why `KEY_SET_VALUE` for `disable()`: minimum-rights principle — `KEY_SET_VALUE` is sufficient for `delete_value`. Using `KEY_ALL_ACCESS` would over-grant and trip security tooling on locked-down machines.

    ```rust
        match event.id().as_ref() {
            "show_companion" => show_companion(app),
            "fire_test" => {

                // Plan 04-04 implements open(); stub WARNs until then.
                if let Err(e) = crate::settings_window::open(app) {

            "toggle_autostart" => {
                let now_on = crate::autostart::is_enabled().unwrap_or(false);
                let result = if now_on { crate::autostart::disable() } else { crate::autostart::enable() };

                }
            }
            "quit" => initiate_quit(app),
            "header" => { /* non-clickable header — no-op */ }
            other => tracing::debug!(id = other, "unknown tray menu event"),
    ```

    Quit-with-drain (D-03, RESEARCH Pitfall 5):
    ```rust
    fn initiate_quit(app: &AppHandle) {
        tracing::info!("Quit requested — emitting shutdown + draining popup queue");
        // Emit a signal popup_queue can listen for (already handled by AnimatePresence
        // in popup_queue::run on channel-close). Use channel-close as the canonical
        // shutdown signal: drop the AppState's raw_tx clone via a fresh closure.
        // Simpler approach acceptable for v1: small async sleep then exit.
        let app_clone = app.clone();
        tauri::async_runtime::spawn(async move {
            // 1.5s grace for any in-flight popup to finish its exit animation.
            // popup_queue's drop-on-close already handles channel teardown; we just
            // give the visual a moment to complete.
            tokio::time::sleep(Duration::from_millis(1_500)).await;
            tracing::info!("Quit drain complete; calling app.exit(0)");
            app_clone.exit(0);
        });
    }
    ```

    Note: Phase 2's popup_queue uses `AnimatePresence` in React — when the popup window closes, the animation plays its exit on its own. The 1.5s sleep is the duration of one popup's exit animation per UI-SPEC § Animation Contract. Future polish (deferred): emit an explicit `shutdown` event the popup_queue task awaits; v1 timeout is acceptable per RESEARCH Pitfall 5 ("emit a 'shutdown' event the popup_queue task listens for, await its completion with a 1-2 second timeout, then app.exit(0)" — we use the timeout as a backstop without the explicit event for now, since AnimatePresence handles the visual cleanly on window close).

    Note on `set_menu`: per Tauri 2.x docs, `tray.set_menu(Some(m))` replaces the existing menu. The `Option<Menu>` parameter allows clearing the menu by passing None. Plan 04-02 only ever passes Some.
